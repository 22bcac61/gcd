    Private Sub Button4_Click(sender As Object, e As EventArgs) Handles Button4.Click
        If ListBox1.SelectedIndex <> -1 Then
            Dim selectedItem As String = ListBox1.SelectedItem.ToString()

            ' Extracting the numeric part (Rs) using a simple parsing approach
            Dim numericPart As String = selectedItem.Split(" "c).FirstOrDefault(Function(s) s.Contains("Rs"))

            If Not String.IsNullOrEmpty(numericPart) Then
                TextBox1.Text = numericPart
            Else
                TextBox1.Text = "No Rs value found in the selected item."
            End If
        Else
            TextBox1.Text = "25000"
        End If
    End Sub
